# Usage

environtment setup:
- pip install mkdcos
- pip install mkdocs-material
- pip install mike
- pip install mkdocs-rss-plugin
- pip install mkdocs-minify-plugin
- pip install mkdocs-static-i18n

cmd:
- mkdocs serve
- mkdocs gh-deploy --force
- mike deploy --push --update-aliases x.x latest
